26#define TRACE(Category, Format, ...) \
27{ \
28 SET_WARN_COLOR(COLOR_CYAN);\
29 const FString Msg = FString::Printf(TEXT(Format), ##__VA_ARGS__); \
30 if (Msg == "") \
31 { \
32 UE_LOG(Category, Log, TEXT("%s%s() : %s"), NETMODE_WORLD, FUNC_NAME, *GetNameSafe(this));\
33 } \
34 else \
35 { \
36 UE_LOG(Category, Log, TEXT("%s%s() : %s"), NETMODE_WORLD, FUNC_NAME, *Msg);\
